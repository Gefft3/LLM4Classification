        <section class="corpo_noticia">
                


                                
    <p class="">
        O Brasil tem atualmente 1942 cidades com risco de desastre ambiental, o que representa quase 35% do total dos municípios brasileiros. O Governo federal fez o mapeamento das regiões suscetíveis a desastres associados a deslizamentos de terras, alagamentos, enxurradas e inundações e concluiu que as áreas dentro dessas quase duas mil cidades consideradas em risco concentram mais de 8,9 milhões de brasileiros, o que representa seis por cento da população nacional. Segundo o mesmo estudo, em 2012, eram 821 cidades em risco e as populações pobres são as mais prováveis de sofrerem com os desastres ambientais no Brasil.

    const contentId = '1827642';
    const contentTypeName = 'Articles';
    const storageKey = `reaction_${contentId}`;

    function setItem(reaction) {
        try {
            localStorage.setItem(storageKey, reaction);
        } catch (e) {
            document.cookie = `${storageKey}=${encodeURIComponent(reaction)};path=/`;
        }
    }

    function getItem() {
        try {
            return localStorage.getItem(storageKey);
        } catch (e) {
            const name = `${storageKey}=`;
            const cookies = document.cookie.split(';');
            for (let cookie of cookies) {
                cookie = cookie.trim();
                if (cookie.startsWith(name)) {
                    return decodeURIComponent(cookie.substring(name.length));
                }
            }
        }
        return null;
    }

    function updateUI(reaction, percentage) {
        $('.valor_feedback_noticia span').text(`${percentage || 0}%`);

        const labelReacao = $('.botao_open_reacao .label_reacao');
        const iconOpenReacao = $('.botao_open_reacao .icon_open_reacao');
        const containerReacoes = $('.container_reacoes');

        const novoSrc = `/i/icon_${reaction.toLowerCase()}.svg`;
        labelReacao.addClass("label_reacao_mobile").html(reaction);

        $('.valor_feedback_noticia').css('visibility', 'initial');

        $('.botao_reacao').hide();

        iconOpenReacao.attr('src', novoSrc);

        const boxShadow = containerReacoes.css('boxShadow');
        containerReacoes.css('boxShadow', (boxShadow === "none" || boxShadow === "0px 0px 0px 0px rgba(0, 0, 0, 0)") ?
            "0 2px 5px 0px rgba(0, 0, 0, .25)" : "none");
    }

    function registerReaction(reaction) {
        CofinaHits.reaction(contentId, contentTypeName, reaction, (data) => {
            setItem(reaction);
            $('.valor_feedback_noticia span').text(`${data.PercentageOfPeopleSameReaction || 0}%`);
        }, console.error);
    }

    $(document).ready(function() {
        const storedReaction = getItem();
        if (storedReaction) {
            CofinaHits.getReactionPercentage(contentId, contentTypeName, storedReaction, (data) => {
                updateUI(storedReaction, data.PercentageOfPeopleSameReaction);
            }, console.error);
        }

        $('.botao_reacao').click(function() {
            const selectedReaction = $(this).attr('rating');
            registerReaction(selectedReaction);
        });
    });

                        $(element).addClass(errorClass).removeClass(validClass);

                        if ($(element).next().hasClass("matriz-ui-combobox")) {
                            $(element).selectpicker("refresh");
                        }
                    },
                    unhighlight: function (element, errorClass, validClass) {
                        $(element).removeClass(errorClass).addClass(validClass);
                        if ($(element).next().hasClass("matriz-ui-combobox")) {
                            $(element).selectpicker("refresh");
                        }
                    }
                };
            }

            function welcomeFormAction(form, event, livechat) {
                event.preventDefault();

                if ($(form).valid()) {
                    introduced = true;

                    hideIntroductionForm();

                    var cpf = $("#ChatWelcomeForm form [name='ChatFormCPF']").val().replace(/[^\d]/g,'');
                    var nome = $("#ChatWelcomeForm form [name='ChatFormNome']").val();
                    var email = $("#ChatWelcomeForm form [name='ChatFormEMail']").val();

                    livechat.registerGuest({
                        "department": "Detran",
                        "name": nome,
                        "email": email
                    });
                    livechat.setCustomField('cpf', cpf);
                    showChat(livechat);
                }
            }

            function initializeRocket() {
                var deferred = $.Deferred();
                if (livechat) {
                    deferred.resolve();
                } else {
                    var href = window.location.href;
                    var url = "https://chat.rs.gov.br";
                    if (href.indexOf(".des.") !== -1) {
                        url = "http://172.28.3.70:3000";
                    }
                    (function (w, d, s, u) {
                        w.RocketChat = function (c) {
                            w.RocketChat._.push(c)
                        };
                        w.RocketChat._ = [];
                        w.RocketChat.url = u;
                        var h = d.getElementsByTagName(s)[0], j = d.createElement(s);
                        j.async = true;
                        j.src = url + '/livechat/rocketchat-livechat.min.js?_=201903270000';
                        h.parentNode.insertBefore(j, h);
                        j.addEventListener("load", function() {
                            livechat = RocketChat.livechat;
                            livechat.hideWidget();
                            deferred.resolve();
                        });
                    })(window, document, 'script', url + '/livechat');
                }
                return deferred.promise();
            }

                $("#ChatWelcomeForm").addClass("hidden");
            }

            function hideChatButtonMenu() {

            }

        });

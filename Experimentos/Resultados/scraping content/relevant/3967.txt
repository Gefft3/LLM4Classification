	function getColor(aqi) {
    if (aqi >= 0 && aqi <= 50) {
        return '#009966'; 
    } else if (aqi > 50 && aqi <= 100) {
        return '#ffde33'; 
    } else if (aqi > 100 && aqi <= 150) {
        return '#f59022'; 
    } else if (aqi > 150 && aqi <= 200) {
        return '#eb2128'; 
    } else if (aqi > 200 && aqi <= 300) {
        return '#a21c4d'; 
    } else {
        return '#8a1b1e'; 
    }
}
